# App-Level Logs Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add structured app-level logging (JSON-line file) and a minimal "Logs" panel in the dashboard so users and maintainers can see what's happening — replacing scattered `console.error` calls and silent `catch {}` blocks.

**Architecture:** A single `logger` module in `packages/core` writes JSON-line entries to `<dataDir>/logs/kontexta.log` with 10MB rolling rotation. A web API route reads recent entries (filterable by source/level/search). A dashboard panel renders them with filters and a 5-second auto-refresh poll. Backend wiring is targeted (highest-signal paths only) — full audit deferred.

**Tech Stack:** TypeScript, Node `fs` (sync append), vitest (core + web tests), Next.js route handlers, React, no new deps.

**Spec:** [`docs/superpowers/specs/2026-05-18-app-logs-design.md`](../specs/2026-05-18-app-logs-design.md)

```typescript
import { appendFileSync, mkdirSync, existsSync, statSync, renameSync, readFileSync } from "node:fs";
import { dirname, join } from "node:path";
import { getDataDir } from "./util/paths.js";

export type LogLevel = "debug" | "info" | "warn" | "error";

export interface LogEntry {
  ts: string;
  level: LogLevel;
  source: string;
  msg: string;
  data?: Record<string, unknown>;
}

export interface Logger {
  debug(msg: string, data?: Record<string, unknown>): void;
  info(msg: string, data?: Record<string, unknown>): void;
  warn(msg: string, data?: Record<string, unknown>): void;
  error(msg: string, data?: Record<string, unknown> | Error): void;
  child(suffix: string): Logger;
}

const LEVELS: Record<LogLevel, number> = { debug: 10, info: 20, warn: 30, error: 40 };
const ROTATE_BYTES = 10 * 1024 * 1024;

function resolveLevel(): LogLevel {
  const v = process.env.KONTEXTA_LOG_LEVEL?.toLowerCase();
  if (v === "debug" || v === "info" || v === "warn" || v === "error") return v;
  return "info";
}

let currentLevel: LogLevel = resolveLevel();

export function setLogLevel(level: LogLevel): void {
  currentLevel = level;
}

function logPath(): string {
  return join(getDataDir(), "logs", "kontexta.log");
}

function rotateIfNeeded(path: string): void {
  try {
    if (!existsSync(path)) return;
    const size = statSync(path).size;
    if (size <= ROTATE_BYTES) return;
    renameSync(path, `${path}.1`);
  } catch {
    // best-effort rotation; ignore failure
  }
}

function serializeData(data: Record<string, unknown> | Error | undefined): Record<string, unknown> | undefined {
  if (!data) return undefined;
  if (data instanceof Error) {
    return { err: { name: data.name, message: data.message, stack: data.stack } };
  }
  return data;
}

function write(entry: LogEntry): void {
  if (LEVELS[entry.level] < LEVELS[currentLevel]) return;
  const path = logPath();
  try {
    mkdirSync(dirname(path), { recursive: true });
    rotateIfNeeded(path);
    appendFileSync(path, JSON.stringify(entry) + "\n");
  } catch {
    // If logging itself fails, swallow — we cannot log the failure to log.
  }
}

function makeLogger(source: string): Logger {
  const mk = (level: LogLevel) => (msg: string, data?: Record<string, unknown> | Error) =>
    write({ ts: new Date().toISOString(), level, source, msg, data: serializeData(data) });
  return {
    debug: mk("debug") as Logger["debug"],
    info: mk("info") as Logger["info"],
    warn: mk("warn") as Logger["warn"],
    error: mk("error"),
    child: (suffix) => makeLogger(`${source}.${suffix}`),
  };
}

export function getLogger(source: string): Logger {
  return makeLogger(source);
}

export interface ReadOpts {
  limit?: number;
  source?: string;
  level?: LogLevel;
  search?: string;
}

function readFileLines(p: string): string[] {
  if (!existsSync(p)) return [];
  return readFileSync(p, "utf-8").split("\n").filter((l) => l.length > 0);
}

function parseEntry(line: string): LogEntry | null {
  try {
    const o = JSON.parse(line);
    if (typeof o?.ts !== "string" || typeof o?.level !== "string" || typeof o?.source !== "string" || typeof o?.msg !== "string") {
      return null;
    }
    return o as LogEntry;
  } catch {
    return null;
  }
}

export function readRecentLogs(opts: ReadOpts = {}): LogEntry[] {
  const limit = Math.min(opts.limit ?? 500, 5000);
  const minLevel = opts.level ? LEVELS[opts.level] : 0;
  const search = opts.search?.toLowerCase();
  const sourceFilter = opts.source;

  const path = logPath();
  const lines = [...readFileLines(path), ...readFileLines(`${path}.1`)];
  // Reverse so newest first; current file is appended chronologically.
  // But .1 is older than current — combine: current (reversed) then .1 (reversed).
  const curLines = readFileLines(path).reverse();
  const oldLines = readFileLines(`${path}.1`).reverse();
  const ordered = [...curLines, ...oldLines];

  const out: LogEntry[] = [];
  for (const line of ordered) {
    const e = parseEntry(line);
    if (!e) continue;
    if (LEVELS[e.level] < minLevel) continue;
    if (sourceFilter && e.source !== sourceFilter && !e.source.startsWith(`${sourceFilter}.`)) continue;
    if (search) {
      const hay = (e.msg + " " + JSON.stringify(e.data ?? {})).toLowerCase();
      if (!hay.includes(search)) continue;
    }
    out.push(e);
    if (out.length >= limit) break;
  }
  return out;
}
```

import { NextResponse } from "next/server";
import { readRecentLogs, type LogLevel } from "kxta-core";
import { checkAuth } from "@/lib/auth";

const VALID_LEVELS = new Set<LogLevel>(["debug", "info", "warn", "error"]);

export async function GET(req: Request) {
  if (!checkAuth(req as any)) {
    return NextResponse.json({ error: "unauthorized" }, { status: 401 });
  }

  const url = new URL(req.url);
  const sp = url.searchParams;

  const limitRaw = sp.get("limit");
  const limit = limitRaw ? Math.max(1, Math.min(5000, parseInt(limitRaw, 10) || 500)) : 500;
  const source = sp.get("source") ?? undefined;
  const search = sp.get("search") ?? undefined;
  const levelRaw = sp.get("level");
  const level = levelRaw && VALID_LEVELS.has(levelRaw as LogLevel) ? (levelRaw as LogLevel) : undefined;

  const entries = readRecentLogs({ limit, source, search, level });
  return NextResponse.json({ entries });
}
```

"use client";

import { useState } from "react";

export interface LogEntry {
  ts: string;
  level: "debug" | "info" | "warn" | "error";
  source: string;
  msg: string;
  data?: Record<string, unknown>;
}

const LEVEL_COLORS: Record<LogEntry["level"], string> = {
  debug: "bg-gray-500/20 text-gray-300",
  info: "bg-blue-500/20 text-blue-300",
  warn: "bg-yellow-500/20 text-yellow-300",
  error: "bg-red-500/20 text-red-300",
};

function relativeTime(iso: string): string {
  const diff = Date.now() - new Date(iso).getTime();
  if (diff < 1000) return "just now";
  if (diff < 60_000) return `${Math.floor(diff / 1000)}s ago`;
  if (diff < 3_600_000) return `${Math.floor(diff / 60_000)}m ago`;
  if (diff < 86_400_000) return `${Math.floor(diff / 3_600_000)}h ago`;
  return new Date(iso).toLocaleDateString();
}

export function LogRow({ entry }: { entry: LogEntry }) {
  const [expanded, setExpanded] = useState(false);
  const hasData = entry.data && Object.keys(entry.data).length > 0;
  return (
    <div className="border-b border-[var(--border)] py-2 px-3 hover:bg-[var(--bg-tertiary)]/50">
      <div className="flex items-baseline gap-2 text-sm">
        <span
          className="text-[var(--text-tertiary)] text-xs w-20 shrink-0 font-mono"
          title={entry.ts}
        >
          {relativeTime(entry.ts)}
        </span>
        <span className={`text-xs px-2 py-0.5 rounded font-mono uppercase shrink-0 ${LEVEL_COLORS[entry.level]}`}>
          {entry.level}
        </span>
        <span className="text-[var(--text-secondary)] text-xs font-mono shrink-0">{entry.source}</span>
        <span className="text-[var(--text-primary)] flex-1 break-all">{entry.msg}</span>
        {hasData && (
          <button
            onClick={() => setExpanded((v) => !v)}
            className="text-xs text-[var(--text-tertiary)] hover:text-[var(--accent)] shrink-0"
          >
            {expanded ? "▾" : "▸"} data
          </button>
        )}
      </div>
      {expanded && hasData && (
        <pre className="mt-2 ml-[6.5rem] p-2 bg-[var(--bg-primary)] border border-[var(--border)] rounded text-xs font-mono overflow-x-auto">
          {JSON.stringify(entry.data, null, 2)}
        </pre>
      )}
    </div>
  );
}
```

```typescript
"use client";

import { useEffect, useMemo, useState } from "react";
import { LogRow, type LogEntry } from "./log-row";

type Level = "debug" | "info" | "warn" | "error";
const ALL_LEVELS: Level[] = ["debug", "info", "warn", "error"];

interface Props {
  onClose: () => void;
}

export function LogsPanel({ onClose }: Props) {
  const [entries, setEntries] = useState<LogEntry[]>([]);
  const [levels, setLevels] = useState<Set<Level>>(new Set(ALL_LEVELS));
  const [source, setSource] = useState("");
  const [search, setSearch] = useState("");
  const [autoRefresh, setAutoRefresh] = useState(false);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // Construct the effective `level` param: highest-deselected determines the min level.
  // Simplest UX: if exactly one level is selected, query that level (and above). If multiple,
  // fetch everything and filter client-side. This avoids implementing a multi-level API.
  const requestUrl = useMemo(() => {
    const sp = new URLSearchParams();
    sp.set("limit", "500");
    if (search.trim()) sp.set("search", search.trim());
    if (source.trim()) sp.set("source", source.trim());
    if (levels.size === 1) sp.set("level", [...levels][0]);
    return `/api/logs?${sp.toString()}`;
  }, [levels, source, search]);

  async function load() {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch(requestUrl);
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const body = await res.json();
      setEntries(Array.isArray(body.entries) ? body.entries : []);
    } catch (e: any) {
      setError(e.message);
    } finally {
      setLoading(false);
    }
  }

  useEffect(() => { load(); }, [requestUrl]);

  useEffect(() => {
    if (!autoRefresh) return;
    const id = setInterval(load, 5000);
    return () => clearInterval(id);
  }, [autoRefresh, requestUrl]);

  const visibleEntries = useMemo(
    () => entries.filter((e) => levels.has(e.level as Level)),
    [entries, levels],
  );

  function toggleLevel(level: Level) {
    setLevels((prev) => {
      const next = new Set(prev);
      if (next.has(level)) next.delete(level); else next.add(level);
      if (next.size === 0) ALL_LEVELS.forEach((l) => next.add(l));
      return next;
    });
  }

  return (
    <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">
      <div className="bg-[var(--bg-secondary)] border border-[var(--border)] rounded-lg w-full max-w-4xl h-[80vh] flex flex-col shadow-2xl">
        <div className="flex items-center justify-between px-4 py-3 border-b border-[var(--border)]">
          <h2 className="text-lg font-bold text-[var(--accent)]">App Logs</h2>
          <button onClick={onClose} aria-label="close" className="text-[var(--text-secondary)] hover:text-[var(--text-primary)]">✕</button>
        </div>

        <div className="px-4 py-2 border-b border-[var(--border)] flex flex-wrap items-center gap-2 text-xs">
          {ALL_LEVELS.map((l) => (
            <button
              key={l}
              onClick={() => toggleLevel(l)}
              className={`px-2 py-1 rounded font-mono uppercase ${levels.has(l) ? "bg-[var(--accent)] text-black" : "bg-[var(--bg-tertiary)] text-[var(--text-secondary)]"}`}
            >
              {l}
            </button>
          ))}
          <input
            type="text"
            value={source}
            onChange={(e) => setSource(e.target.value)}
            placeholder="source filter (e.g. mcp)"
            className="bg-[var(--bg-primary)] border border-[var(--border)] rounded px-2 py-1 text-xs w-48 font-mono"
          />
          <input
            type="text"
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            placeholder="search…"
            className="bg-[var(--bg-primary)] border border-[var(--border)] rounded px-2 py-1 text-xs flex-1 min-w-[8rem]"
          />
          <label className="flex items-center gap-1 cursor-pointer select-none">
            <input type="checkbox" checked={autoRefresh} onChange={(e) => setAutoRefresh(e.target.checked)} className="accent-[var(--accent)]" />
            <span className="text-[var(--text-secondary)]">auto-refresh (5s)</span>
          </label>
          <button onClick={load} className="px-2 py-1 rounded bg-[var(--bg-tertiary)] text-[var(--text-secondary)] hover:text-[var(--text-primary)]">
            {loading ? "…" : "refresh"}
          </button>
        </div>

        <div className="flex-1 overflow-y-auto">
          {error && (
            <div className="p-3 text-red-400 text-sm">Failed to load logs: {error}</div>
          )}
          {!error && visibleEntries.length === 0 && !loading && (
            <div className="p-6 text-center text-[var(--text-tertiary)] text-sm">No entries match the current filters.</div>
          )}
          {visibleEntries.map((e, i) => (
            <LogRow key={`${e.ts}-${i}`} entry={e} />
          ))}
        </div>

        <div className="px-4 py-2 border-t border-[var(--border)] text-xs text-[var(--text-tertiary)] flex justify-between">
          <span>Showing {visibleEntries.length} of {entries.length} loaded</span>
          <a href="/api/logs?limit=5000" download="kontexta.log.json" className="hover:text-[var(--accent)]">Download recent</a>
        </div>
      </div>
    </div>
  );
}
```

# kxta-core

Internal library used by both the MCP server (`apps/mcp`) and the dashboard (`apps/web`).

## Logging

Modules can write structured logs via the logger module:

\`\`\`typescript
import { getLogger } from "kxta-core";
const log = getLogger("my.module");
log.info("something happened", { count: 5 });
log.error("operation failed", err);
\`\`\`

Logs are written as JSON lines to `<dataDir>/logs/kontexta.log` with 10MB rolling rotation (one backup). The global threshold is `info` by default, configurable via the `KONTEXTA_LOG_LEVEL` env var (`debug` | `info` | `warn` | `error`).

**Sensitive data:** module authors are responsible for not logging credentials, tokens, or PII. There is no automatic redaction.
